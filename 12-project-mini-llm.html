01 第一步：训练自己的 Tokenizer

第 1 章讲了 BPE。这里我们用 HuggingFace 的 tokenizers 库，从零训练一个 BPE 分词器：

train_tokenizer.py —— 从零训练 BPE Tokenizer
"""
train_tokenizer.py —— 训练自己的 BPE 分词器
这就是第 1 章 BPE 算法的工业实现
"""
from tokenizers import Tokenizer
from tokenizers.models import BPE
from tokenizers.trainers import BpeTrainer
from tokenizers.pre_tokenizers import ByteLevel

# ① 创建字节级 BPE 模型（第 1 章：任何字符都能还原到字节，永不 OOV）
tokenizer = Tokenizer(BPE(unk_token="<unk>"))
tokenizer.pre_tokenizer = ByteLevel(add_prefix_space=True)

# ② 训练器：词表大小、特殊 token
trainer = BpeTrainer(
    vocab_size=2000,              # 小词表（演示用，真实 10w+）
    special_tokens=["<pad>", "<unk>", "<s>", "</s>"],
)

# ③ 在语料上训练
tokenizer.train(files=["data/corpus.txt"], trainer=trainer)
tokenizer.save("tinylm-tokenizer.json")
print("✅ Tokenizer 训练完成")

# 试一下
enc = tokenizer.encode("大模型真有趣")
print(f"分词: {enc.tokens}")      # ['大', '模', '型', '真', '有', '趣']
print(f"IDs: {enc.ids}")

02 第二步：准备预训练数据

大模型吃 token 序列。我们要把文本变成模型能消化的格式：

prepare_data.py —— 数据准备
"""
prepare_data.py —— 把文本语料转成预训练用的 token 序列
"""
from tokenizers import Tokenizer
import numpy as np

tokenizer = Tokenizer.from_file("tinylm-tokenizer.json")

def prepare_corpus(text_path, out_path, context_len=64):
    """把整篇语料切成固定长度的训练样本"""
    text = open(text_path).read()
    # 整体编码成一个长 token 序列
    ids = tokenizer.encode(text).ids
    print(f"语料共 {len(ids)} 个 token")

    # 切成 context_len 长度的样本（这就是第 1 章的上下文窗口）
    samples = []
    for i in range(0, len(ids) - context_len, context_len):
        samples.append(ids[i:i+context_len])
    np.save(out_path, np.array(samples))
    print(f"✅ 生成 {len(samples)} 个训练样本，保存到 {out_path}")

prepare_corpus("data/corpus.txt", "data/train.npy")

03 第三步：搭建模型（带 RoPE）

第 1 章的 MiniGPT 用了简单位置编码。这里升级为带 RoPE 的完整 Transformer，更接近真实模型：

tinylm/model.py —— 带 RoPE 的迷你 GPT
"""
tinylm/model.py —— TinyLM 模型：带 RoPE 的 Decoder-Only Transformer
约 200 万参数，结构上与 GLM-5.2 同源
"""
import math, torch
import torch.nn as nn
import torch.nn.functional as F

# ---------- RoPE 旋转位置编码（第 1 章第 8 节的数学实现） ----------
def precompute_rope(head_dim, max_seq=512, base=10000):
    inv_freq = 1.0 / (base ** (torch.arange(0, head_dim, 2).float() / head_dim))
    t = torch.arange(max_seq).float()
    freqs = torch.outer(t, inv_freq)
    return torch.cos(freqs), torch.sin(freqs)

def apply_rope(x, cos, sin):
    # x: (B, T, n_head, head_dim)，对最后一维做旋转
    x1, x2 = x[..., ::2], x[..., 1::2]
    rotated = torch.stack([x1 * cos - x2 * sin, x1 * sin + x2 * cos], dim=-1)
    return rotated.flatten(-2)

# ---------- Attention（带 RoPE + 因果掩码 + KV Cache） ----------
class Attention(nn.Module):
    def __init__(self, d_model, n_head):
        super().__init__()
        self.n_head = n_head
        self.head_dim = d_model // n_head
        self.qkv = nn.Linear(d_model, 3 * d_model, bias=False)
        self.proj = nn.Linear(d_model, d_model, bias=False)
        self.register_buffer("cos", torch.zeros(512, self.head_dim // 2))
        self.register_buffer("sin", torch.zeros(512, self.head_dim // 2))

    def forward(self, x, kv_cache=None):
        B, T, C = x.shape
        qkv = self.qkv(x).reshape(B, T, 3, self.n_head, self.head_dim)
        q, k, v = qkv.permute(2, 0, 3, 1, 4)  # (3,B,n_head,T,head_dim)

        # 应用 RoPE（旋转位置编码）
        q = apply_rope(q, self.cos[:T], self.sin[:T])
        k = apply_rope(k, self.cos[:T], self.sin[:T])

        # KV Cache：推理时拼接历史（第 1 章第 5 节）
        if kv_cache is not None:
            k = torch.cat([kv_cache["k"], k], dim=3)
            v = torch.cat([kv_cache["v"], v], dim=3)

        # 因果注意力
        attn = (q @ k.transpose(-2, -1)) / math.sqrt(self.head_dim)
        mask = torch.tril(torch.ones(T, k.shape[3], device=x.device))
        attn = attn.masked_fill(mask == 0, float("-inf"))
        attn = F.softmax(attn, dim=-1)
        out = (attn @ v).transpose(1, 2).reshape(B, T, C)
        return self.proj(out), {"k": k, "v": v}

# ---------- Transformer Block（Pre-Norm 残差结构） ----------
class Block(nn.Module):
    def __init__(self, d_model, n_head):
        super().__init__()
        self.attn = Attention(d_model, n_head)
        self.ln1 = nn.LayerNorm(d_model)
        self.mlp = nn.Sequential(
            nn.Linear(d_model, 4*d_model), nn.GELU(), nn.Linear(4*d_model, d_model))
        self.ln2 = nn.LayerNorm(d_model)
    def forward(self, x, kv_cache=None):
        a, kv = self.attn(self.ln1(x), kv_cache)
        x = x + a
        x = x + self.mlp(self.ln2(x))
        return x, kv

# ---------- 完整 TinyLM ----------
class TinyLM(nn.Module):
    def __init__(self, vocab=2000, d_model=128, n_head=4, n_layer=4):
        super().__init__()
        self.tok_emb = nn.Embedding(vocab, d_model)
        self.blocks  = nn.ModuleList([Block(d_model, n_head) for _ in range(n_layer)])
        self.ln_f    = nn.LayerNorm(d_model)
        self.head    = nn.Linear(d_model, vocab, bias=False)
        # 共享权重（减少参数，真实模型常用）
        self.head.weight = self.tok_emb.weight
        print(f"参数量: {sum(p.numel() for p in self.parameters())/1e6:.1f}M")

    def forward(self, idx, targets=None):
        x = self.tok_emb(idx)
        for block in self.blocks:
            x, _ = block(x)
        x = self.ln_f(x)
        logits = self.head(x)
        if targets is None: return logits, None
        loss = F.cross_entropy(logits.view(-1, logits.size(-1)), targets.view(-1))
        return logits, loss

04 第四步：预训练（就是下一个 token 预测）

pretrain.py —— 预训练循环
"""
pretrain.py —— 预训练：让 TinyLM 学会预测下一个 token
这就是第 2 章讲的「预训练」的最小实现
"""
import torch, numpy as np
from tinylm.model import TinyLM

device = "cuda" if torch.cuda.is_available() else "cpu"
data = torch.from_numpy(np.load("data/train.npy")).long().to(device)
print(f"训练数据: {data.shape}, 设备: {device}")

model = TinyLM().to(device)
opt = torch.optim.AdamW(model.parameters(), lr=3e-4, weight_decay=0.1)
sched = torch.optim.lr_scheduler.CosineAnnealingLR(opt, T_max=1000)

BATCH, STEPS = 64, 1000
CTX = data.shape[1]
for step in range(STEPS):
    # 随机采样 batch
    ix = torch.randint(0, len(data), (BATCH,))
    batch = data[ix]
    x, y = batch[:, :-1], batch[:, 1:]      # 错开一位：预测下一个

    logits, loss = model(x, y)
    opt.zero_grad()
    loss.backward()
    torch.nn.utils.clip_grad_norm_(model.parameters(), 1.0)  # 梯度裁剪（第 2 章稳定性）
    opt.step()
    sched.step()

    if step % 100 == 0:
        print(f"step {step:4d}  loss {loss.item():.4f}  lr {sched.get_last_lr()[0]:.2e}")

torch.save(model.state_dict(), "tinylm-pretrained.pt")
print("✅ 预训练完成，模型已保存")

跑起来你会看到 loss 从 ~7.5（随机猜测 2000 词表）一路下降到 ~2-3（开始学到语言规律）。这就是第 2 章那张「训练 loss 曲线」的真实模样，只是规模小了十万倍。

05 第五步：SFT 对齐（让它会对话）

预训练后的模型只会「续写」，不会「回答」。SFT 让它学会问答格式：

sft.py —— 监督微调
"""
sft.py —— SFT：让模型学会「问-答」格式
用少量高质量对话数据微调（第 2 章第 5 节）
"""
import torch, json
from tinylm.model import TinyLM

# SFT 数据：指令-回答对（第 2 章第 5.1 节的格式）
SFT_DATA = [
    {"q": "你好", "a": "你好！我是 TinyLM，有什么可以帮你？"},
    {"q": "1+1等于几", "a": "1+1 等于 2。"},
    {"q": "什么是大模型", "a": "大模型是用海量数据训练的神经网络，能理解和生成语言。"},
    # ... 更多数据
]

def sft_train(model, tokenizer, epochs=20):
    opt = torch.optim.AdamW(model.parameters(), lr=1e-4)
    for ep in range(epochs):
        total = 0
        for item in SFT_DATA:
            # 拼成 "问：{q}\n答：{a}" 格式
            text = f"问：{item['q']}\n答：{item['a']}"
            ids = tokenizer.encode(text).ids
            if len(ids) < 5: continue
            x = torch.tensor([ids[:-1]]).to(model.tok_emb.weight.device)
            y = torch.tensor([ids[1:]]).to(model.tok_emb.weight.device)
            _, loss = model(x, y)
            opt.zero_grad(); loss.backward(); opt.step()
            total += loss.item()
        print(f"epoch {ep}  avg_loss {total/len(SFT_DATA):.3f}")
    torch.save(model.state_dict(), "tinylm-sft.pt")
    print("✅ SFT 完成")

06 第六步：推理生成（带 KV Cache）

generate.py —— 带采样与 KV Cache 的生成
"""
generate.py —— 文本生成（带温度采样 + KV Cache 加速）
"""
import torch
from tokenizers import Tokenizer

@torch.no_grad()
def generate(model, tokenizer, prompt, max_new=50, temperature=0.8, top_k=20):
    model.eval()
    ids = tokenizer.encode(prompt).ids
    x = torch.tensor([ids]).to(model.tok_emb.weight.device)

    for _ in range(max_new):
        logits, _ = model(x)                    # 前向（生产可加 KV Cache）
        logits = logits[:, -1, :] / temperature  # 取最后一个位置
        if top_k:
            v, _ = torch.topk(logits, top_k)    # Top-K 采样
            logits[logits < v[:, [-1]]] = -float("inf")
        probs = torch.softmax(logits, dim=-1)
        next_id = torch.multinomial(probs, num_samples=1)
        x = torch.cat([x, next_id], dim=1)

    return tokenizer.decode(x[0].tolist())

# ---------- 试用你亲手训的模型 ----------
model = TinyLM().to(device)
model.load_state_dict(torch.load("tinylm-sft.pt"))
tokenizer = Tokenizer.from_file("tinylm-tokenizer.json")

while True:
    q = input("你: ")
    if q == "exit": break
    print("TinyLM:", generate(model, tokenizer, f"问：{q}\n答："))

注意：架构和算法几乎一样！TinyLM 和 GLM-5.2 的核心都是「Transformer + 预测下一个 token + SFT」。差距主要在规模、数据、工程优化上。这就是为什么「理解 TinyLM = 理解大模型的骨架」——剩下的是把每个维度做到极致的工程艺术。
