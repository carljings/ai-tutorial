01 去重：MinHash 与 SimHash

重复数据让模型「死记」而非「泛化」，必须去除。精确去重（完全相同）用哈希，近似去重（高度相似）用 MinHash。

dedup.py —— MinHash 近似去重
"""
dedup.py —— MinHash + LSH 近似去重
处理百万级文档，找出相似度 > 阈值的文档对
"""
from datasketch import MinHash, MinHashLSH
import re

def shingles(text, k=5):
    """把文本切成 k-shingle（k 个词一组）"""
    words = re.findall(r'\w+', text.lower())
    return [" ".join(words[i:i+k]) for i in range(len(words)-k+1)]

def make_minhash(text, num_perm=128):
    """生成 MinHash 签名"""
    m = MinHash(num_perm=num_perm)
    for s in shingles(text):
        m.update(s.encode('utf8'))
    return m

# 建 LSH 索引（局部敏感哈希，加速近似搜索）
lsh = MinHashLSH(threshold=0.8, num_perm=128)
docs = {"doc1": "大模型是当代最重要的技术", "doc2": "大模型是当代最关键的技术", "doc3": "今天天气不错"}

for doc_id, text in docs.items():
    mh = make_minhash(text)
    lsh.insert(doc_id, mh)

# 查询：找 doc1 的近似重复
result = lsh.query(make_minhash(docs["doc1"]))
print(result)  # ['doc1', 'doc2'] —— doc2 被识别为近似重复

02 质量过滤：多级漏斗

quality_filter.py —— 多级质量过滤
"""
quality_filter.py —— 数据质量过滤流水线
启发式规则 + 小模型打分 + 去乱码
"""
import re

class QualityFilter:
    def __init__(self):
        self.min_chars = 50
        self.max_repeat = 0.3   # 重复字符最大占比

    def filter(self, text):
        # ① 长度过滤
        if len(text) < self.min_chars: return False
        # ② 去乱码（非文字占比过高）
        alpha_ratio = sum(c.isalpha() or '\u4e00'<=c<='\u9fff' for c in text) / len(text)
        if alpha_ratio < 0.5: return False
        # ③ 重复检测
        chars = list(text)
        repeat_ratio = 1 - len(set(chars)) / len(chars)
        if repeat_ratio > self.max_repeat: return False
        # ④ 去 HTML/模板残留
        if '<' in text and '>' in text: return False
        # ⑤ 困惑度过滤（用小语言模型，低困惑度=高质量）
        # ppl = small_lm.perplexity(text)  # 生产用
        # if ppl > threshold: return False
        return True

qf = QualityFilter()
print(qf.filter("这是一段足够长的正常中文文本，质量过关。"))  # True
print(qf.filter("啊啊啊啊啊啊啊啊啊啊"))                    # False（重复）

pii_redact.py —— 个人信息脱敏
"""pii_redact.py —— 识别并脱敏个人信息"""
import re

PATTERNS = {
    "phone":   (re.compile(r'1[3-9]\d{9}'),           "[手机]"),
    "id_card": (re.compile(r'\d{17}[\dXx]'),          "[身份证]"),
    "email":   (re.compile(r'[\w.-]+@[\w.-]+\.\w+'),  "[邮箱]"),
    "bank":    (re.compile(r'\d{16,19}'),              "[卡号]"),
}

def redact_pii(text):
    for _, (pattern, replacement) in PATTERNS.items():
        text = pattern.sub(replacement, text)
    return text

print(redact_pii("联系 13812345678 或 test@gmail.com"))
# "联系 [手机] 或 [邮箱]"

04 数据配比：玄学也是科学

GLM-5 的关键经验：训练初期就大幅增加代码与推理数据。配比设计原则：

  • 代码 30-40%：塑造推理能力骨架；
  • 网页 30-40%：广覆盖的常识与知识；
  • 书籍/论文 10-15%：高质量长文本；
  • 数学/科学 5-10%：逻辑推理；
  • 对话 5%：对话格式；
  • 配比不是固定的：训练不同阶段动态调整（课程学习）。

05 持续监控

  • 数据漂移：新批次数据的分布是否偏离预期；
  • 质量回归：新数据质量是否下降；
  • 毒性/偏见检测：有害内容比例是否超标；
  • 去重率监控：去重比例是否异常（可能过滤太狠）。

★ MinHash 去重的工程优化

前面的 MinHash 示例是「教学版」，真要处理十亿级文档会遇到两个瓶颈：内存爆炸（每篇文档存 128 个 4 字节哈希，10 亿篇就是 500GB）和逐条插入太慢。工业界有三招优化。第一招是降低 num_perm——128 位精度对应阈值 0.8，但若只想去掉「几乎一模一样」的文档（阈值 0.9+），64 位甚至 32 位就够了，内存直接减半再减半。第二招是MinHash LSH 的 band 分桶，它把签名切成多段（band），只有落在同一段的文档才进候选集，把两两比较从 O(n²) 降到近 O(n)。第三招是分片 + 并行，用 Spark/Dask 把文档按哈希分到不同节点，各自去重后再合并。

"""minhash_opt.py —— 大规模 MinHash 去重的工程优化版"""
from datasketch import MinHash, MinhashLSH
import hashlib, re

# 优化 1：用更省内存的 MinHash（num_perm 按阈值定）
# 阈值 0.9 → 64 位足够；想抓更宽松的相似(0.7)才需要 128+
NUM_PERM = 64

def fast_shingles(text, k=7):
    # 优化 2：直接对字符级 n-gram 哈希，省去分词开销
    text = re.sub(r"\s+", "", text.lower())  # 去空白，对中文友好
    return {hashlib.md5((text[i:i+k]).encode()).hexdigest() for i in range(len(text)-k+1)}

def make_mh(shingles):
    m = MinHash(num_perm=NUM_PERM)
    for s in shingles:
        m.update(s.encode())
    return m

# 优化 3：LSH 自动做 band 分桶，查询只比候选，不全量两两比对
lsh = MinhashLSH(threshold=0.9, num_perm=NUM_PERM, weights=(0.5, 0.5))

def dedup_stream(doc_iter):
    """流式去重：来一篇查一次，命中重复就丢弃，省内存"""
    kept = 0
    for doc_id, text in doc_iter:
        mh = make_mh(fast_shingles(text))
        if lsh.query(mh):        # 已有相似文档 → 判为重复
            continue
        lsh.insert(doc_id, mh)
        kept += 1
        yield doc_id, text
    print(f"去重完成，保留 {kept} 篇")

# 生产中再叠一层：按 doc_id 哈希分 100 个分片并行，每个分片独立去重
# shard = int(hashlib.md5(doc_id.encode()).hexdigest(), 16) % 100

这套优化在 Common Crawl 级别的语料（数百 GB）上能把去重从「跑几天」压到「几小时」。一个常被忽视的细节是 weights=(b, r) 参数：它控制「高召回 vs 高精度」的权衡。去重场景通常宁可误杀也不漏，所以偏向高召回（b 大、r 小）；如果是查重/推荐，则反过来要高精度。

★ 数据质量评分：从规则到模型

前面质量过滤用的是「过/不过」的硬阈值。更精细的做法是给每篇文档打一个连续的质量分，下游可以按分数采样、按分数加权——高质量文档多采、低质量文档少采。下面这段代码把长度、重复率、困惑度、信息密度四个维度归一化后加权求和，输出 0~1 的质量分。它可以直接接进数据配比模块，让「玄学配比」变成「数据驱动的配比」。

"""quality_score.py —— 多维数据质量打分器"""
import math, re

def entropy_rate(text):
    """字符熵率：信息密度的代理指标，越高信息量越大"""
    freq = {}
    for ch in text:
        freq[ch] = freq.get(ch, 0) + 1
    n = len(text)
    return -sum((c/n) * math.log2(c/n) for c in freq.values())

def quality_score(text, ppl=None):
    # 维度 1：长度分（50~2000 字为佳，过短/过长扣分）
    L = len(text)
    len_score = 1.0 if 50 <= L <= 2000 else min(L/50, 2000/max(L,1))

    # 维度 2：重复率分（字符多样性越高越好）
    repeat = 1 - len(set(text)) / max(len(text), 1)
    rep_score = max(0, 1 - repeat * 3)

    # 维度 3：困惑度分（需接小语言模型，这里允许为空）
    ppl_score = 1.0
    if ppl is not None:
        ppl_score = math.exp(-max(ppl - 20, 0) / 60)   # ppl=20 满分，越大越扣

    # 维度 4：信息密度（熵率归一化，中文上限约 8 bit）
    info_score = min(entropy_rate(text) / 6.0, 1.0)

    # 加权融合（权重可按下游任务调）
    score = 0.25*len_score + 0.25*rep_score + 0.30*ppl_score + 0.20*info_score
    return round(score, 3)

# 用法：按分数决定采样权重
for doc in corpus:
    s = quality_score(doc)
    weight = s ** 2          # 平方让高质量文档被采样的概率更高
    # sampler.maybe_keep(doc, weight)

这种「评分 + 加权采样」是 GLM、Llama 等大模型预训练数据清洗的主流做法。一个实战经验：困惑度权重给到 0.3 是有原因的——它是区分「人写的好文章」和「机器生成的低质文本」最有效的单一信号，因为语言模型对自己生成的东西困惑度天然偏低，而高质量人类文本往往落在 15~40 的「甜区」。把这个维度做扎实，能挡掉大量合成数据和 SEO 农场文本。

★ 数据配比的消融实验

配比到底怎么影响模型？光靠直觉不行，得做消融实验。下面这张表是一次典型的配比消融——固定总 token 数和训练步数，只改各来源的比例，看下游 benchmark 的变化。结论非常清晰：代码数据是推理能力的发动机（加代码 GSM8K 涨 8 个点），但代码过多会伤语言理解（MMLU 掉）；适量的书籍/论文提升长文能力；对话数据少了模型不会聊天，多了又变啰嗦。配比的本质，是在多种能力间做帕累托权衡。

★ 数据污染：评测集泄露的红线

数据工程里有一条看不见但致命的红线——评测集污染。如果你训练数据里混进了 MMLU、GSM8K 这些 benchmark 的题目和答案，模型在评测时会「假性」飙高，但真实能力毫无提升。这会让你的所有评估失真，进而做出完全错误的决策（比如以为微调有效其实是在背题）。所以严肃的数据流水线都必须有一道「去污染」工序：把训练语料和评测集做严格匹配，命中就剔除。

"""decontaminate.py —— 训练数据去评测集污染"""
import re, json
from datasketch import MinHash, MinHashLSH

def normalize(text):
    # 统一小写、去标点空白，让匹配更鲁棒
    return re.sub(r"[^\w]", "", text.lower())

def build_benchmark_index(bench_items, num_perm=64):
    # 把所有评测题建成 LSH 索引，用于快速查重
    lsh = MinHashLSH(threshold=0.5, num_perm=num_perm)  # 阈值低，宁可误杀
    for i, item in enumerate(bench_items):
        mh = MinHash(num_perm=num_perm)
        for ngram in [normalize(item)[j:j+8] for j in range(len(normalize(item))-8)]:
            mh.update(ngram.encode())
        lsh.insert(i, mh)
    return lsh

def decontaminate(train_docs, lsh, num_perm=64):
    clean = []
    for doc in train_docs:
        mh = MinHash(num_perm=num_perm)
        norm = normalize(doc)
        for j in range(len(norm)-8):
            mh.update(norm[j:j+8].encode())
        if lsh.query(mh):       # 和某道评测题疑似重复 → 剔除
            continue
        clean.append(doc)
    print(f"去污染：{len(train_docs)} → {len(clean)} 篇")
    return clean

# 实战：评测集越多越要查（MMLU/GSM8K/HumanEval/C-Eval 都要进索引）
# 经验：Common Crawl 里约 1-3% 的文档会和某 benchmark 高度相似，必须剔除

去污染的阈值要故意设低（0.5 而非 0.8），因为宁可误删一些无害训练数据，也绝不能放过一道泄露的考题——前者损失一点数据量，后者直接毁掉整个评估的可信度。除了显式 benchmark，还要警惕「隐性污染」：某些热门博客抄了 GSM8K 的题目做讲解，这种间接泄露用上面的 n-gram 匹配也能抓到。一个专业的预训练团队，去污染这道工序的严谨程度，往往决定了它发布的 benchmark 分数能不能被同行认可——这也是为什么顶会论文越来越要求附「去污染证明」。

深 数据质量度量：从“脏数据”到“可用数据”

“数据是新代码”这句话的潜台词是：数据质量决定模型上限。但质量不是玄学，它可以被度量、被监控、被回归。下面给出几个可落地的数据质量度量维度和监控代码。

数据清洗的“去重 + 去毒 + 去低质”三件套

高质量预训练数据集（如 RedPajama、SlimPajama）都经过严格的去重和过滤。下面是最小化的数据清洗流水线。

import re, hashlib
from collections import defaultdict

def clean_text(t):
    t = re.sub(r"<[^>]+>", " ", t)            # 去 HTML 标签
    t = re.sub(r"\s+", " ", t).strip()        # 规范空白
    return t

def quality_score(t):
    """低质文本检测：长度、重复率、符号比的简单启发式。"""
    if len(t) < 50: return 0.0                 # 太短
    words = t.split()
    rep = 1 - len(set(words)) / max(len(words),1)  # 重复词比例
    sym = sum(1 for c in t if not c.isalnum() and not c.isspace()) / max(len(t),1)
    score = 1.0 - 0.5*rep - 3.0*sym             # 重复和符号过多都扣分
    return max(0.0, min(1.0, score))

def dedup(docs):
    seen, out = set(), []
    for d in docs:
        # MinHash/局部敏感哈希才是生产级方案，这里用精确前缀哈希示意
        h = hashlib.md5(clean_text(d)[:200].encode()).hexdigest()
        if h not in seen:
            seen.add(h); out.append(d)
    return out

raw = open("corpus.txt").read().splitlines()
cleaned = [clean_text(d) for d in raw if quality_score(clean_text(d)) > 0.5]
deduped = dedup(cleaned)
print(f"原始 {len(raw)} → 清洗后 {len(cleaned)} → 去重后 {len(deduped)}")

生产级去重要用 MinHash + LSH（局部敏感哈希）检测近似重复，而非精确哈希——因为“换了几个字”的近重复才是大头。质量打分也要用训练好的分类器，而非纯启发式。

质量维度度量方法危害处理
重复MinHash/LSH 近似去重模型过拟合重复模式删除或降采样
毒性毒性分类器输出有害内容过滤或人工审核
低质质量分类器/启发式拉低整体能力过滤
PII(隐私)正则+NER检测合规风险脱敏或删除
配比失衡领域分布统计偏科重新采样平衡

数据质量的持续监控

数据是“活的”——持续有新数据进来，质量会漂移。所以数据质量必须持续监控而非一次性清洗。做法是建立数据质量 dashboard：每次新增数据，自动跑去重率、毒性率、领域分布、平均质量分等指标，与历史基线对比，异常（如毒性率突增）触发告警。微调场景更要监控标注一致性——多人标注同一批数据时，计算标注者间一致率(Kappa)，低于 0.7 说明标注规范不清，得先统一标准再继续。记住：垃圾进、垃圾出，而且垃圾往往很隐蔽，唯有度量能让它现形。Chinchilla 定律告诉我们模型能力与数据量成对数关系，但前提是数据质量恒定——低质数据再多也无济于事。

Q1：训练数据去重，精确去重还是模糊去重？

必须模糊去重（MinHash+LSH）。精确去重只能抓完全相同的，但「换了几个字」的近重复才是大头（占 20~40%）。近重复数据会让模型过拟合这些模式、浪费训练算力。生产级去重流程：先精确去重（快，去完全相同）→ 再 MinHash+LSH 近似去重（去近重复）→ 最后文档级和段落级都要做。还有跨数据集去重（不同来源数据间也可能重复）。去重是数据清洗里 ROI 最高的一步，能显著提升模型质量。
