03 第一步：搭建 Harness 骨架

Harness 是 CodeMate 的地基。我们整合第 3 章的权限、Hook、上下文装配，成一个统一的 Harness 类：

codemate/harness/__init__.py —— Harness 整合
"""
codemate/harness/__init__.py —— Harness 整合层
把权限、Hook、上下文、成本控制组合起来
"""
from .permissions import PermissionEngine, Decision
from .hooks import HookManager, HookEvent
from .context import ContextAssembler
from .cost import CostTracker

class Harness:
    """CodeMate 的工程外壳：权限 + Hook + 上下文 + 成本"""
    def __init__(self, config):
        self.permissions = PermissionEngine(config["rules"])
        self.hooks       = HookManager()
        self.context     = ContextAssembler(config["context_budget"])
        self.cost        = CostTracker(config["budget"])
        self._register_default_hooks()

    def _register_default_hooks(self):
        # 默认 Hook：审计日志 + 危险拦截 + 写后格式化
        self.hooks.register(HookEvent.PRE_TOOL, self._audit_log)
        self.hooks.register(HookEvent.PRE_TOOL, self._block_danger)
        self.hooks.register(HookEvent.POST_TOOL, self._auto_format)

    def _audit_log(self, ctx):
        import logging
        logging.info(f"[工具] {ctx['tool']}({ctx.get('args','')[:50]})")
        return ctx

    def _block_danger(self, ctx):
        if "rm -rf" in str(ctx.get("args", "")):
            return None   # 拦截
        return ctx

    def _auto_format(self, ctx):
        if ctx["tool"] == "write_file" and ctx.get("path", "").endswith(".py"):
            ctx["feedback"] = "建议跑 black 格式化"
        return ctx

    def check_tool(self, action):
        """检查一个工具调用是否被允许"""
        return self.permissions.check(action)

    def before_tool(self, ctx):
        return self.hooks.run(HookEvent.PRE_TOOL, ctx)

    def after_tool(self, ctx):
        return self.hooks.run(HookEvent.POST_TOOL, ctx)

    def can_spend(self, tokens):
        """检查预算是否还够"""
        return self.cost.can_spend(tokens)
codemate/harness/cost.py —— 成本追踪
"""codemate/harness/cost.py —— 成本与预算控制"""
class CostTracker:
    def __init__(self, budget_tokens=500000):
        self.budget = budget_tokens
        self.used   = 0
        self.calls  = 0

    def record(self, tokens):
        self.used  += tokens
        self.calls += 1

    def can_spend(self, estimated):
        return self.used + estimated <= self.budget

    def report(self):
        return {"used": self.used, "budget": self.budget,
                "calls": self.calls,
                "ratio": round(self.used/self.budget*100, 1)}

04 第二步：代码库 RAG 索引

CodeMate 要「懂」你的项目，得先把整个代码库索引成可检索的知识。这是第 6 章的实战应用——而且针对代码有特殊处理：

codemate/knowledge/indexer.py —— 代码库索引
"""
codemate/knowledge/indexer.py —— 把代码库索引成可检索的知识
特殊处理：按函数/类切分，保留符号元数据
"""
import os, ast
from sentence_transformers import SentenceTransformer
import numpy as np
import chromadb

embedder = SentenceTransformer("BAAI/bge-m3")

class CodeIndexer:
    def __init__(self, project_root):
        self.root = project_root
        self.db = chromadb.PersistentClient(path=".codemate/db")
        self.collection = self.db.get_or_create_collection("code")

    def index_project(self):
        """遍历项目，把每个函数/类索引成一个 chunk"""
        chunks = []
        for dirpath, _, files in os.walk(self.root):
            for f in files:
                if not f.endswith(".py"): continue
                path = os.path.join(dirpath, f)
                if ".codemate" in path: continue
                chunks.extend(self._index_python_file(path))

        # 批量 embedding + 入库
        if chunks:
            texts  = [c["content"] for c in chunks]
            embs   = embedder.encode(texts).tolist()
            self.collection.add(
                ids=[c["id"] for c in chunks],
                documents=texts, embeddings=embs,
                metadatas=[{"file": c["file"], "symbol": c["symbol"],
                            "type": c["type"]} for c in chunks])
        print(f"✅ 索引完成：{len(chunks)} 个代码符号")

    def _index_python_file(self, path):
        """用 AST 把 Python 文件按函数/类切分（第 6 章代码切分）"""
        rel = os.path.relpath(path, self.root)
        try:
            tree = ast.parse(open(path).read())
        except:
            return []
        chunks = []
        for node in ast.walk(tree):
            if isinstance(node, (ast.FunctionDef, ast.AsyncFunctionDef)):
                seg = ast.get_source_segment(open(path).read(), node)
                if seg and len(seg) > 20:
                    chunks.append({"id": f"{rel}:{node.name}:{node.lineno}",
                                   "content": f"# 文件: {rel}\n" + seg,
                                   "file": rel, "symbol": node.name, "type": "function"})
            elif isinstance(node, ast.ClassDef):
                seg = ast.get_source_segment(open(path).read(), node)
                if seg and len(seg) > 20:
                    chunks.append({"id": f"{rel}:{node.name}:{node.lineno}",
                                   "content": f"# 文件: {rel}\n" + seg[:2000],
                                   "file": rel, "symbol": node.name, "type": "class"})
        return chunks

注意几个针对代码的优化：按函数/类切分（而非固定长度），保留语法完整性；元数据带文件名和符号名，检索结果可溯源；用 AST 而非正则，准确可靠。这就是第 6 章「代码感知切分」的生产级实现。

05 第三步：核心 Agent Loop

整合 Harness、RAG、工具，写出 CodeMate 的核心循环。这是全书第 4 章的工业级落地：

codemate/agent/loop.py —— CodeMate 核心 Loop
"""
codemate/agent/loop.py —— CodeMate 的核心 Agent Loop
整合 Harness（权限/Hook/成本）+ RAG 检索 + 工具调用 + PEV
"""
import json
from openai import OpenAI
from ..harness import Harness, Decision

client = OpenAI(base_url="https://open.bigmodel.cn/api/paas/v4", api_key="YOUR_KEY")
MODEL = "glm-5.2"

SYSTEM = """你是 CodeMate，企业级 AI 编程助手。工作原则：
1. 改代码前先读相关文件，理解上下文；
2. 每次改动后跑测试验证（PEV 工作法）；
3. 测试失败时分析错误、修正、重试；
4. 涉及写操作前说明意图；
5. 工具失败如实报告，不臆测；
6. 完成后给出简短总结与改动清单。"""

class CodeMate:
    def __init__(self, harness: Harness, retriever, skill_registry, tools):
        self.harness = harness
        self.retriever = retriever          # RAG 检索器
        self.skills = skill_registry         # 技能注册表
        self.tools = tools                   # 工具实现

    def run(self, task, max_steps=25):
        # ① 装配初始上下文：系统提示 + RAG 检索 + 技能匹配
        relevant_code = self.retriever.search(task, top_k=5)
        skills = self.skills.discover(task, top_k=2)
        skill_text = "\n".join(s["workflow"] for s in skills)

        messages = [
            {"role": "system", "content": SYSTEM + f"\n\n相关代码：\n{relevant_code}\n\n可用技能：\n{skill_text}"},
            {"role": "user", "content": task},
        ]

        last_actions = []
        for step in range(max_steps):
            # 预算检查
            if not self.harness.can_spend(4000):
                return "⚠️ 预算耗尽"

            # 任务漂移检查（每 5 步）
            if step % 5 == 0 and step > 0:
                messages.append({"role": "system",
                    "content": f"提醒：原任务是「{task}」，确认是否在正轨。"})

            try:
                resp = client.chat.completions.create(model=MODEL, messages=messages,
                                                     tools=self._tool_defs())
                self.harness.cost.record(resp.usage.total_tokens)
                msg = resp.choices[0].message
                messages.append(msg)

                if not msg.tool_calls:
                    return msg.content    # 完成

                for tc in msg.tool_calls:
                    obs = self._execute_tool(tc, last_actions)
                    messages.append({"role": "tool", "tool_call_id": tc.id, "content": obs})

            except Exception as e:
                messages.append({"role": "system", "content": f"系统错误：{e}，请调整策略。"})

        return "达到最大步数"

    def _execute_tool(self, tc, last_actions):
        name = tc.function.name
        args = json.loads(tc.function.arguments)
        action = f"{name}:{args}"

        # 死循环检测
        if last_actions[-3:].count(action) >= 2:
            return "[系统] 检测到重复动作，请换思路"
        last_actions.append(action)

        # 权限检查（第 3 章）
        decision = self.harness.check_tool(action)
        if decision.decision == Decision.DENY:
            return f"[权限拒绝] {decision.reason}"
        if decision.decision == Decision.ASK:
            if input(f"⚠️ 确认 {action}? [y/N]: ").lower() != "y":
                return "[用户拒绝]"

        # PreToolUse Hook
        pre = self.harness.before_tool({"tool": name, "args": args})
        if pre is None:
            return "[Hook 拦截]"

        # 执行（带重试）
        result = self._call_with_retry(name, args)

        # PostToolUse Hook（PEV 的 V：验证 + 反馈）
        post = self.harness.after_tool({"tool": name, "result": result, "args": args})
        if post and post.get("feedback"):
            result += f"\n[反馈] {post['feedback']}"
        return result[:3000]

    def _call_with_retry(self, name, args, retries=2):
        for i in range(retries + 1):
            try:
                return self.tools[name](**args)
            except Exception as e:
                if i == retries: return f"[错误] {e}"
        return "[未知错误]"

这个 Loop 是全书所有理念的熔炉：RAG 检索（第 6 章）注入项目知识、技能发现（第 8 章）加载合适 Skill、Harness（第 3 章）管权限/Hook/成本、Agent Loop（第 4 章）的多重防护（死循环、漂移、预算、重试）、PEV 环（写后跑测试）。读懂这 100 行，你就读懂了 Claude Code 的核心。

import os

def read_file(path, max_chars=10000):
    """读取文件内容"""
    if not os.path.exists(path): return f"[错误] 文件不存在: {path}"
    with open(path) as f:
        content = f.read(max_chars)
    return content if len(content) < max_chars else content + "\n...[截断]"

def write_file(path, content):
    """写入文件（自动创建目录）"""
    os.makedirs(os.path.dirname(path) or ".", exist_ok=True)
    with open(path, "w") as f:
        f.write(content)
    return f"✅ 已写入 {path} ({len(content)} 字符)"

def list_dir(path="."):
    """列出目录"""
    if not os.path.isdir(path): return f"[错误] 不是目录: {path}"
    items = []
    for name in sorted(os.listdir(path)):
        full = os.path.join(path, name)
        tag = "📁" if os.path.isdir(full) else "📄"
        items.append(f"{tag} {name}")
    return "\n".join(items[:50])
codemate/tools/shell.py —— Shell 命令工具
"""codemate/tools/shell.py —— 执行 shell 命令"""
import subprocess

def run_shell(command, timeout=30):
    """执行 shell 命令，返回输出（PEV 的关键：能跑测试）"""
    try:
        r = subprocess.run(command, shell=True, capture_output=True,
                          text=True, timeout=timeout)
        out = (r.stdout + r.stderr)[:3000]
        return f"[exit {r.returncode}]\n{out}" or "[无输出]"
    except subprocess.TimeoutExpired:
        return "[超时]"
    except Exception as e:
        return f"[错误] {e}"

07 第五步：内置技能（SKILL.md）

CodeMate 的「聪明」很大程度来自内置技能。每个技能是一个 Markdown 文件，描述「这类任务怎么做」：

codemate/skills/builtin/testing.md —— 「写测试」技能
---
name: write-tests
description: 为指定函数或模块编写单元测试
trigger: ["写测试", "加单元测试", "补测试", "test coverage"]
tools: [read_file, write_file, run_shell]
---
# 写测试技能

## 工作流程
1. **理解目标**：读目标函数，理清输入输出、边界条件、异常情况
2. **查规范**：看项目里现有测试的风格（pytest？unittest？mock 用法？）
3. **设计用例**：
   - 正常路径（happy path）至少 2 个
   - 边界值（空、零、最大、最小）
   - 异常输入（错误类型、越界）
   - 如果有外部依赖，用 mock
4. **写测试文件**：放到对应的 tests/ 目录，命名 test_xxx.py
5. **跑测试**：`pytest tests/test_xxx.py -v`
6. **PEV 循环**：失败就分析、修正、重跑，直到全绿

## 经验（tips）
- 一个测试只测一件事，函数名说明测的是什么
- 用 fixture 复用 setup
- 别测私有方法，只测公共接口
- 覆盖率不是目的，测对关键路径才是

---
name: debug
description: 定位并修复 bug
trigger: ["修 bug", "报错", "不工作", "崩溃", "调试"]
tools: [read_file, run_shell, list_dir]
---
# 调试技能

## 工作流程
1. **复现**：先拿到能稳定复现报错的最小步骤
2. **读错误**：看 traceback，定位文件和行号
3. **读上下文**：读报错处及调用的函数
4. **提出假设**：可能的原因是什么？（类型错？空值？逻辑错？）
5. **验证假设**：加 print/日志，或跑最小复现
6. **修复**：改最小必要代码
7. **回归测试**：跑相关测试确认没引入新问题

## 经验
- 不要在没复现前就改代码
- 一次只改一处，改完就测
- 如果修了 30 分钟没进展，重新读错误信息，可能方向错了
- 修复后想想：这个 bug 的根因是什么？别处有没有类似问题？

08 第六步：CLI 入口

codemate/cli.py —— 命令行入口
"""
codemate/cli.py —— CodeMate 命令行入口
用法：python -m codemate "给 utils.py 加测试"
     或交互模式：python -m codemate
"""
import sys, argparse
from .harness import Harness
from .knowledge.indexer import CodeIndexer
from .knowledge.retriever import CodeRetriever
from .skills.registry import SkillRegistry
from .agent.loop import CodeMate
from .tools.filesystem import read_file, write_file, list_dir
from .tools.shell import run_shell

CONFIG = {
    "rules": [
        ("read_file:**", "allow"), ("list_dir:**", "allow"),
        ("run_shell:git status", "allow"), ("run_shell:pytest*", "allow"),
        ("write_file:**", "ask"),
        ("run_shell:rm*", "deny"), ("run_shell:git push*", "deny"),
    ],
    "context_budget": 100000,
    "budget": 500000,
}

TOOLS = {"read_file": read_file, "write_file": write_file,
         "list_dir": list_dir, "run_shell": run_shell}

def main():
    parser = argparse.ArgumentParser(description="CodeMate - AI 编程助手")
    parser.add_argument("task", nargs="?", help="任务描述")
    parser.add_argument("--reindex", action="store_true", help="重新索引代码库")
    args = parser.parse_args()

    # 初始化各层
    harness = Harness(CONFIG)
    indexer  = CodeIndexer(".")
    if args.reindex or not indexer.collection.count():
        print("📚 索引代码库...")
        indexer.index_project()
    retriever = CodeRetriever(indexer.collection)
    skills    = SkillRegistry("codemate/skills/builtin")

    mate = CodeMate(harness, retriever, skills, TOOLS)

    if args.task:
        # 单次任务模式
        print(mate.run(args.task))
    else:
        # 交互模式
        print("🤖 CodeMate 已就绪（输入 exit 退出）")
        while True:
            task = input("\n❯ ").strip()
            if task in ("exit", "quit"): break
            if task: print(mate.run(task))

    print(f"\n📊 本次成本：{harness.cost.report()}")

if __name__ == "__main__":
    main()
