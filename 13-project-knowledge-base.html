实战：从零搭建企业知识库
一个端到端的智能问答系统

企业最迫切的 AI 需求之一就是「让 AI 回答基于公司文档的问题」。这一章我们用第 6-7 章的 RAG + Wiki 知识，搭一个生产级的企业知识库：支持多格式文档、混合检索、引用溯源、管理后台、评估闭环。这是能直接拿去落地的项目。

阅读约 40 分钟 难度 ★★★★ 含完整代码 可直接落地

★ 业务场景与需求

假设我们在为一家科技公司搭建内部知识库「AskCorp」。需求：

  • 员工用自然语言提问，系统基于公司文档（产品手册、HR 制度、技术规范、会议纪要…）回答；
  • 回答必须有引用（来自哪份文档第几页），可溯源、可信任；
  • 支持多格式（PDF、Word、Markdown、网页）；
  • 文档可增量更新，不用每次重建全库；
  • 有管理后台上传/删除文档、查看问答日志；
  • 有评估闭环，持续优化检索质量。

02 文档解析与索引流水线

这是知识库的「建库」环节，第 6 章两条流水线的离线部分。要支持多格式：

askcorp/ingest.py —— 多格式文档解析与索引
"""
askcorp/ingest.py —— 文档解析 + 分块 + 索引
支持 PDF / Word / Markdown / TXT
"""
import hashlib, os
from langchain_community.document_loaders import (
    PyPDFLoader, Docx2txtLoader, UnstructuredMarkdownLoader, TextLoader)
from langchain_text_splitters import RecursiveCharacterTextSplitter
import chromadb
from sentence_transformers import SentenceTransformer

embedder = SentenceTransformer("BAAI/bge-m3")
splitter = RecursiveCharacterTextSplitter(chunk_size=400, chunk_overlap=60,
    separators=["\n\n", "\n", "。", ".", " ", ""])

LOADERS = {".pdf": PyPDFLoader, ".docx": Docx2txtLoader,
           ".md": UnstructuredMarkdownLoader, ".txt": TextLoader}

class IngestPipeline:
    def __init__(self, db_path=".askcorp/db"):
        self.client = chromadb.PersistentClient(path=db_path)
        self.collection = self.client.get_or_create_collection("docs",
            metadata={"hnsw:space": "cosine"})

    def ingest(self, file_path, source_meta=None):
        ext = os.path.splitext(file_path)[1].lower()
        loader_cls = LOADERS.get(ext)
        if not loader_cls: raise ValueError(f"不支持的格式: {ext}")

        # ① 解析
        docs = loader_cls(file_path).load()
        text = "\n".join(d.page_content for d in docs)

        # ② 分块（第 6 章递归切分）
        chunks = splitter.split_text(text)

        # ③ 索引：每个块带完整元数据（可溯源的关键）
        doc_id = hashlib.md5(file_path.encode()).hexdigest()[:8]
        embs = embedder.encode(chunks).tolist()
        metadatas = [{
            "doc_id": doc_id, "source": os.path.basename(file_path),
            "chunk_idx": i, "total": len(chunks),
            **(source_meta or {}),
        } for i in range(len(chunks))]

        self.collection.add(
            ids=[f"{doc_id}_{i}" for i in range(len(chunks))],
            documents=chunks, embeddings=embs, metadatas=metadatas)
        print(f"✅ {file_path} → {len(chunks)} 个块已索引")
        return doc_id, len(chunks)

    def delete_doc(self, doc_id):
        """增量删除：按 doc_id 删除某文档的所有块"""
        self.collection.delete(where={"doc_id": doc_id})
        print(f"🗑️ 已删除文档 {doc_id}")

关键设计：元数据带 doc_id/source/chunk_idx，这是引用溯源的基础；增量删除按 doc_id 批量删，文档更新时不用重建全库；用 cosine 距离（与归一化后的 BGE 向量匹配）。

03 混合检索 + Rerank

askcorp/retrieve.py —— 检索层
"""
askcorp/retrieve.py —— 混合检索 + Rerank + 查询改写
第 6 章检索理论的落地
"""
from rank_bm25 import BM25Okapi
from sentence_transformers import CrossEncoder

reranker = CrossEncoder("BAAI/bge-reranker-v2-m3")

class Retriever:
    def __init__(self, collection, embedder):
        self.collection = collection
        self.embedder = embedder

    def rewrite_query(self, query, chat_history, llm):
        # 查询改写：消解指代（第 6 章第 5.1 节）
        if not chat_history: return query
        rewritten = llm(f"根据对话历史，把最后的问题改写得更完整（只输出改写后的问题）："
                        f"\n历史：{chat_history[-3:]}\n问题：{query}")
        return rewritten.strip()

    def search(self, query, top_k=5):
        # ① 向量检索（语义）
        q_emb = self.embedder.encode([query]).tolist()
        vec_hits = self.collection.query(query_embeddings=q_emb, n_results=20)

        # ② 简单关键词补充（演示用，生产可加 BM25）
        candidates = vec_hits["documents"][0]
        metas = vec_hits["metadatas"][0]

        # ③ Rerank 精排（第 6 章第 5.3 节）
        if candidates:
            scores = reranker.predict([(query, c) for c in candidates])
            ranked = sorted(zip(scores, candidates, metas), key=lambda x: -x[0])
            return [{"content": c, "score": float(s), "meta": m}
                    for s, c, m in ranked[:top_k]]
        return []

04 带引用的答案生成

askcorp/generate.py —— 带引用的答案生成
"""
askcorp/generate.py —— 基于检索结果生成带引用的答案
关键是：让模型只用提供的资料，并标注引用编号
"""
from zhipuai import ZhipuAI
client = ZhipuAI()

PROMPT = """你是企业知识助手。严格根据下面的资料回答问题。

规则：
1. 只用资料里的信息，资料没有就说"根据现有资料无法回答"
2. 不要编造、不要猜测
3. 在引用信息处标注 [编号]，编号对应资料序号
4. 回答要简洁、准确、有条理

资料：
{context}

问题：{question}

请回答："""

def answer(question, retrieved):
    # 拼装带编号的上下文
    context_parts = []
    for i, r in enumerate(retrieved, 1):
        m = r["meta"]
        context_parts.append(f"[{i}] 来源：{m.get('source','未知')}（块 {m.get('chunk_idx','?')}）\n{r['content']}")
    context = "\n\n".join(context_parts)

    resp = client.chat.completions.create(model="glm-5.2", messages=[
        {"role": "user", "content": PROMPT.format(context=context, question=question)}])
    return resp.choices[0].message.content, retrieved

注意 prompt 里的四条规则：只用资料、不编造、标引用、简洁。这是抑制幻觉、保证可溯源的关键。提示词工程在 RAG 系统里和检索质量同等重要。

05 把它变成 API 服务

askcorp/api.py —— FastAPI 服务
"""
askcorp/api.py —— FastAPI 服务：问答 + 文档管理
"""
from fastapi import FastAPI, UploadFile, File
from pydantic import BaseModel
from .ingest import IngestPipeline
from .retrieve import Retriever
from .generate import answer

app = FastAPI(title="AskCorp 企业知识库")
pipeline = IngestPipeline()
retriever = Retriever(pipeline.collection, pipeline.embedder)

class Query(BaseModel):
    question: str
    history: list = []

@app.post("/ask")
def ask(q: Query):
    # 完整流水线：改写 → 检索 → 生成
    query = retriever.rewrite_query(q.question, q.history, llm=lambda x: x)  # 简化
    retrieved = retriever.search(query, top_k=5)
    ans, refs = answer(q.question, retrieved)
    return {"answer": ans, "references": refs}

@app.post("/upload")
def upload(file: UploadFile = File(...)):
    path = f"/tmp/{file.filename}"
    open(path, "wb").write(file.file.read())
    doc_id, n = pipeline.ingest(path)
    return {"doc_id": doc_id, "chunks": n}

@app.delete("/doc/{doc_id}")
def delete(doc_id: str):
    pipeline.delete_doc(doc_id)
    return {"status": "ok"}

06 评估闭环：持续优化

知识库上线后，必须有评估闭环。用 RAGAS 思路自建评估：

askcorp/eval.py —— 评估脚本
"""
askcorp/eval.py —— 评估检索与回答质量
关键指标：检索召回率、答案忠实度、答案相关性
"""
TEST_SET = [
    {"q": "年假怎么算？", "expected_source": "员工手册", "expected_answer_contains": "工龄"},
    {"q": "报销流程是什么？", "expected_source": "财务制度", "expected_answer_contains": "审批"},
    # ... 标几十条真实问题
]

def evaluate(retriever, answer_fn):
    results = []
    for t in TEST_SET:
        retrieved = retriever.search(t["q"], top_k=5)
        ans, _ = answer_fn(t["q"], retrieved)
        # 检索召回：期望的文档是否被检索到
        sources = [r["meta"].get("source", "") for r in retrieved]
        recall = t["expected_source"] in " ".join(sources)
        # 答案忠实：是否包含期望关键词
        faithful = t["expected_answer_contains"] in ans
        results.append({"q": t["q"], "recall": recall, "faithful": faithful})
    # 汇总
    n = len(results)
    print(f"检索召回率: {sum(r['recall'] for r in results)/n:.1%}")
    print(f"答案忠实率: {sum(r['faithful'] for r in results)/n:.1%}")
    return results
📊 检索召回率

该查到的文档，查到了吗？调分块/Embedding/Rerank。

🎯 答案忠实率

答案对不对？调 prompt/生成模型。

⚠️ 持续回归

每次改系统跑评估，防止退化。
